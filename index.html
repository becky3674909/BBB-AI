    function exportScheduleICS() {

        const now = new Date();

        const startOfWeek = new Date(now);

        startOfWeek.setDate(now.getDate() - now.getDay());

        startOfWeek.setHours(0,0,0,0);

        

        const endOfNextWeek = new Date(startOfWeek);

        endOfNextWeek.setDate(startOfWeek.getDate() + 13);

        endOfNextWeek.setHours(23,59,59,999);


        let filtered = scheduleItems.filter(item => {

            const itemDate = new Date(`${item.date}T${item.time || '00:00'}`);

            return itemDate >= startOfWeek && itemDate <= endOfNextWeek;

        });


        if (filtered.length === 0) return alert('本週/下週沒有行程可匯出！');


        let icsContent = `BEGIN:VCALENDAR\nVERSION:2.0\nPRODID:-//BBB AI Assistant//EN`;

        

        filtered.forEach(item => {

            const startStr = item.date.replace(/-/g, '') + 'T' + item.time.replace(':', '') + '00';

            let endDate = new Date(`${item.date}T${item.time}`);

            endDate.setHours(endDate.getHours() + 1);

            const endMonth = String(endDate.getMonth() + 1).padStart(2, '0');

            const endDay = String(endDate.getDate()).padStart(2, '0');

            const endHour = String(endDate.getHours()).padStart(2, '0');

            const endMin = String(endDate.getMinutes()).padStart(2, '0');

            const endStr = `${endDate.getFullYear()}${endMonth}${endDay}T${endHour}${endMin}00`;


            icsContent += `\nBEGIN:VEVENT\nDTSTART:${startStr}\nDTEND:${endStr}\nSUMMARY:${item.title}\nEND:VEVENT`;

        });


        icsContent += `\nEND:VCALENDAR`;


        const blob = new Blob([icsContent], { type: 'text/calendar;charset=utf-8' });

        const link = document.createElement('a'); 

        link.href = window.URL.createObjectURL(blob);

        link.setAttribute('download', `本週下週行程.ics`); 

        document.body.appendChild(link); 

        link.click(); 

        document.body.removeChild(link);

    }


    function calcBudget() {

        const rate = parseFloat(document.getElementById('exchange-rate').value) || 0;

        const totalAud = parseFloat(document.getElementById('budget-total').value) || 0;

        const spentAud = parseFloat(document.getElementById('budget-spent').value) || 0;

        document.getElementById('budget-total-twd').innerText = `≈ NT$ ${Math.round(totalAud * rate).toLocaleString()}`;

        document.getElementById('budget-spent-twd').innerText = `≈ NT$ ${Math.round(spentAud * rate).toLocaleString()}`;

        document.getElementById('budget-remaining').innerText = `剩餘預算: A$ ${(totalAud - spentAud).toLocaleString()}`;

        saveData();

    }

    async function loadData() {

        let dataToLoad = null;


        // 嘗試從雲端讀取

            try {

                const response = await fetch(SCRIPT_URL);

                dataToLoad = await response.json();

                document.getElementById('sync-status').innerText = "✅ 已載入雲端資料";

            } catch (e) {

                console.error('雲端讀取失敗', e);

                document.getElementById('sync-status').innerText = "⚠️ 雲端失敗，使用設備舊資料";

            }

        } else {

            document.getElementById('sync-status').innerText = "⚠️ 尚未設定雲端網址";

        }


        // 如果雲端沒資料或讀取失敗，改用本地舊資料

        if (!dataToLoad) {

            const saved = localStorage.getItem(STORAGE_KEY);

            if (saved) {

                dataToLoad = JSON.parse(saved);

            }

        }


        // 應用資料到畫面

        if (dataToLoad) {

            document.getElementById('today-notes').value = dataToLoad.todayNotes || ''; 

            document.getElementById('speak-days').value = dataToLoad.speakDays || 0;

            // 舊資料相容轉換

            houseItems = dataToLoad.houseItems || (dataToLoad.house ? [{id: Date.now(), text: dataToLoad.house}] : []);

            travelItems = dataToLoad.travelItems || (dataToLoad.travel ? [{id: Date.now()+1, text: dataToLoad.travel}] : []);

            

            if (dataToLoad.countdowns) countdowns = dataToLoad.countdowns;

            if (dataToLoad.scheduleItems) {

                scheduleItems = dataToLoad.scheduleItems;

            } else {

                scheduleItems = [...DEFAULT_SCHEDULE];

            }

            if (dataToLoad.hub) dataToLoad.hub.forEach((v, i) => { let el = document.getElementById(`hub-sel-${i}`); if(el) { el.value = v; updateSelectColor(el); } });

            if (dataToLoad.yoga) dataToLoad.yoga.forEach((v, i) => { let el = document.getElementById(`yoga-sel-${i}`); if(el) { el.value = v; updateSelectColor(el); } });

            if (dataToLoad.green) dataToLoad.green.forEach((v, i) => { let el = document.getElementById(`green-sel-${i}`); if(el) { el.value = v; updateSelectColor(el); } });

            whvItems = dataToLoad.whvList || [];
